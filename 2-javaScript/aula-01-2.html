// existem comportamentos estranhos envolvendo declaração de variaveis como no exemplo abaixo

console.log(a)

// caso eu rode somente isso haverá um erro total

var a = 'eu existo'

// já neste caso, não haverá erro, mas sim, mostrará que a variável não foi definida - 'undefined'. Como se a variável
// 'a' já tivesse sido declarada antes do console.log. Ou seja, não respeita a ordem de leitura do codigo. Esse
// comportamento atípico do javascript é chamado d hoisting.

// com o let, não acontece nada disso. dá um erro total se não for declarado a variável

// vamos a um exemplo com o let

if (true) {
    let b = 'estou aqui dentro da função'
    console.log(b)
    //aqui ele vai imprimir
}
